# =============================================================================
# INSTALLATION NOTES FOR UBUNTU SERVER:
# =============================================================================
# 
# 1. Create virtual environment:
#    python3 -m venv venv
#    source venv/bin/activate
#
# 2. Upgrade pip:
#    pip install --upgrade pip setuptools wheel
#
# 3. Install requirements:
#    pip install -r requirements.txt
#
# 4. Set up PostgreSQL database:
#    sudo -u postgres createuser primus
#    sudo -u postgres createdb primus_db -O primus
#    sudo -u postgres psql -c "ALTER USER primus PASSWORD 'your_password';"
#
# 5. Configure environment variables in .env file
#
# 6. Run database migrations:
#    alembic upgrade head
#
# 7. Start with Gunicorn (production):
#    gunicorn -w 4 -k uvicorn.workers.UvicornWorker main:app --bind 0.0.0.0:8000
#
# 8. Configure Nginx as reverse proxy
#
# 9. Set up SSL with Let's Encrypt:
#    sudo certbot --nginx -d yourdomain.com
#
# =============================================================================
